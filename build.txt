# Build Qt with "configure -webengine-proprietary-codecs"
# https://wiki.qt.io/Building_Qt_6_from_Git

mkdir qt6-build
cd qt6-build
..\qt6\configure -webengine-proprietary-codecs -release -prefix ../install
cmake --build .
cmake --install .

# Build ReeePlayer

mkdir build
cd "build"
set CMAKE_PREFIX_PATH=C:\Qt\6.2.2\install
cmake ..\
cmake --build . --config Release
cd ..

# Build pyutils

cd pyutils
build.bat
cd ..

# Prepare install folder

xcopy "build\ReeePlayer\Release\ReeePlayer.exe" install\
xcopy "build\Launcher\Release\ReeePlayerLauncher.exe" install\
xcopy pyutils\dist\pyutils\* install\pyutils\
